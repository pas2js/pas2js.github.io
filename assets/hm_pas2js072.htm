procedure WriteLnIntRange(const value : Integer);
begin
   case value of
      -2..-1     : WriteLn('negative');
      0          : WriteLn('zero');
      5, 10      : WriteLn('multiple of five');
      1..4, 6..9 : WriteLn('between 1 and 9 but 5');
   else
      WriteLn('elsewhere');
   end;
end;
