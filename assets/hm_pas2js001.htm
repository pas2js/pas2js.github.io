unit Unit1;

interface

uses
  SmartCL.System;

type
  TApplication  = class
    procedure RunApp;
  end;

implementation

{ TApplication }
procedure TApplication.RunApp;
begin
  WriteLn('starting app');
end;
end.


var TObject = {
   $ClassName : "TObject",
   $Parent : null,
   ClassName : function (a) {
      return a.$ClassName;
   },
   ClassType : function (a) {
      return a;
   },
   ClassParent : function (a) {
      return a.$Parent;
   },
   $Init : function () {},
   Create : function (a) {
      return a;
   },
   Destroy : function (a) {
      for (var b in a) {
         a.hasOwnProperty(b) && delete a.prop;
      }
   },
   Destroy$ : function (a) {
      return a.ClassType.Destroy(a);
   },
   Free : function (a) {
      null !== a && a.ClassType.Destroy(a);
   }
};
function $New(a) {
   var b = {
      ClassType : a
   };
   a.$Init(b);
   return b;
}
var TApplication = {
   $ClassName : "TApplication",
   $Parent : TObject,
   $Init : function (a) {
      TObject.$Init(a);
   },
   RunApp : function () {
      window.console && window.console.log("starting app");
   },
   Destroy : TObject.Destroy
}, Application = null, Application = TObject.Create($New(TApplication));
TApplication.RunApp(Application);
