function NumericToMyOptions(n: integer): TMyOptions;

var

  Op: TMyOption;

begin

  Result:= [];

  for Op:= Low(TMyOption) to High(TMyOption) do

    if n and (1 shl ord(Op)) > 0 then Include(Result, Op);

end;
