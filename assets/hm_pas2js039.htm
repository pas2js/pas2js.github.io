type

  end;

procedure WriteLnArray(a: array of TRec);
var
  i: Integer;
begin
  Writeln(IntToStr(Length(a)) + ':{');
  for i := 0 to High(a) do
  begin
    if i > 0 then
      Writeln(',');
    Writeln(IntToStr(a[i].Field));
  end;
  Writeln('}');
end;
