procedure WriteLnArray(a: array of Integer);
var
  i: Integer;
begin
  Writeln(IntToStr(Length(a)) + ':{');
  for i := 0 to High(a) do
  begin
    if i > 0 then
      Writeln(',');
    Writeln(IntToStr(a[i]));
  end;
  Writeln('}');
end;
