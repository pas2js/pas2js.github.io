function MakeArray(n: Integer): array of Integer;
begin
  Result := New Integer[n];
  while n > 0 do
  begin
    Dec(n);
    Result[n] := n;
  end;
end;

procedure WriteLnArray(a: array of Integer);
  var  i: Integer;
begin
  Writeln(IntToStr(a.Length) + ' elements');
  for i := a.Low to a.High do
  begin
    Writeln(i);
    Writeln(': ');
    Writeln(a[i]);
  end;
end;

begin
a := MakeArray(0);
a.Reverse;

function $NewArray(n, v) {
	var r = new Array(n);
	for (var i = 0; i < n; i++)
		r[i] = v;
	return r
}

	var t = a[i1];
	a[i1] = a[i2];
